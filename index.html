Book Club Setup
Connect Supabase to get started — free & takes ~5 minutes
  1. Go to supabase.com → sign up free
  2. Click New project — name it "bookclub"
  3. Go to Authentication → Providers → ensure Email is enabled
  4. Go to Authentication → Email Templates → disable "Confirm email" so accounts work instantly
  5. Go to SQL Editor and run the SQL below
  6. Go to Project Settings → API, copy your Project URL and anon key
-- Run this in Supabase SQL Editor create table if not exists club_settings ( key text primary key, value text ); create table if not exists members ( id uuid primary key, email text unique not null, name text not null, role text default 'user', avatar_color text default '#c47c2b', created_at timestamptz default now() ); create table if not exists proposed_books ( id text primary key, title text not null, author text not null, pages integer, status text default 'pending', suggested_by uuid references members(id), created_at timestamptz default now() ); create table if not exists votes ( id text primary key, book_id text references proposed_books(id) on delete cascade, member_id uuid references members(id), created_at timestamptz default now(), unique(book_id, member_id) ); create table if not exists history ( id text primary key, title text not null, author text not null, pages integer, read_date text, meetup_date text, created_at timestamptz default now() ); -- Row level security (open for anon key) alter table club_settings enable row level security; alter table members enable row level security; alter table proposed_books enable row level security; alter table votes enable row level security; alter table history enable row level security; create policy "all_club_settings" on club_settings for all using (true) with check (true); create policy "all_members" on members for all using (true) with check (true); create policy "all_proposed_books" on proposed_books for all using (true) with check (true); create policy "all_votes" on votes for all using (true) with check (true); create policy "all_history" on history for all using (true) with check (true); -- Enable realtime alter publication supabase_realtime add table club_settings; alter publication supabase_realtime add table proposed_books; alter publication supabase_realtime add table votes; alter publication supabase_realtime add table history;
